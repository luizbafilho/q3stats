def parse
  @log_lines.each do |line|
    if game? line
      Event.new(:game, captures)
    elsif player? line
      Event.new(:player, captures)
    elsif kill? line
      Event.new(:kill, captures)
    end.tap{ |event|
      @events << event if event
    }
  end
  @events
end
