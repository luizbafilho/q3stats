def add_kill raw_kill
  @total_kills += 1
  p = player_by_name raw_kill.first
  if p
    p.add_kill
  else
    killed_player = player_by_name raw_kill[1]
    killed_player.remove_kill
  end
end

def add_player player
  p = player_by_id(player.id)
  if !p
    @players << player
  elsif p.name != player.name
    p.name = player.name
  end
end
